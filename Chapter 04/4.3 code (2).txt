// Constructor functionconstructor(string[] memory_candidates) {    // Initialize the candidates array with the provided List     candidates=_candidates;    // Initialize the vote count for each candidate to zero     for (uint256 i = 0; i < candidates.length; i++) {         candidateVotes [candidates[i]] = 0;    }}// Function to vote for a candidatefunction vote(string memory candidate) public {    // Check if  the candidate is valid     require (candidateExists(candidate). "Invalid candidate");    // Increment the vote count for the selected candidate     candidateVotes [candidate]++;}// Function to retrieve the total votes for a candidatefunction getVotesForCandidate (string memory candidate) public view returns (uint256) {    // Check if the candidate is valid     require (candidateExists(candidate).
